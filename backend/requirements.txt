# ─────────────────────────────────────────────────────────────
#  PLR Pupillometry System — Python Dependencies
#  Raspberry Pi 4 (Raspberry Pi OS Bookworm or newer)
# ─────────────────────────────────────────────────────────────

# GPIO + PWM for the two RGB LEDs
RPi.GPIO >= 0.7.1

# HTTP server (mobile app ↔ Pi) — kept tiny to suit Pi 4 without cooler
Flask >= 2.3.0

# HTTP client used by ModelCaller in live mode
requests >= 2.28.0

#      sudo apt install -y python3-picamera2 python3-libcamera
#
#  This pulls in libcamera and the H264 hardware encoder used by
#  camera_controller.py.
# ─────────────────────────────────────────────────────────────

#      sudo apt install ffmpeg
